function add(x, y) {
  return x + y;
}
console.log(add(2, 3)); // 输出：5

const subtract = function(x, y) {
  return x - y;
};

console.log(subtract(5, 3)); // 输出：2

const multiply = (x, y) => x * y;
console.log(multiply(3, 4)); // 输出：12
